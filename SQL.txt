CREATE TABLE USER(
ID INT NOT NULL auto_increment,
NAME CHAR(20),
  PRIMARY KEY  (`ID`)

);

CREATE TABLE CHALLENGES(
ID INT NOT NULL auto_increment,
NAME CHAR(20),
DESCRIPTION CHAR(20),
CATEGORY CHAR(20),
POINTS INT,
PRIMARY KEY  (`ID`)
);

CREATE TABLE J_USER_CHALLENGES(
ID INT NOT NULL auto_increment,
USERID INT,
CHALLENGE INT,
CHALLENGER INT,
ACTIVE BIT(1),
NOTIFIED BIT(1),
COMPLETED BIT(1),
S_TIME TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
E_TIME TIMESTAMP,
PRIMARY KEY ('ID'),
FOREIGN KEY(USERID) REFERENCES USER(ID),
FOREIGN KEY(CHALLENGE) REFERENCES CHALLENGES(ID),
FOREIGN KEY(CHALLENGER) REFERENCES USER(ID)

);
